import AuthService from "../services/auth.service";

 setMessage("");
    setLoading(true);

    AuthService.login(email, password).then(
      () => {
        props.history.push("/");
        window.location.reload();
      },
      (error) => {
        const resMessage =
          (error.response &&
            error.response.data &&
            error.response.data.message) ||
          error.message ||
          error.toString();

        setLoading(false);
        setMessage(resMessage);
      }
    );
  };
